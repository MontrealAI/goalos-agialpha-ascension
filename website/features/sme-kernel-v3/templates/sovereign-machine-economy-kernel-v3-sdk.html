Reference implementation

A real adapter stays inside its authority.

enterprise-evaluator-adapter.jsKernel v3 contract
class EnterpriseEvaluatorAdapter extends AdapterContract {
  async initialize(context) { return this.bindPolicy(context); }
  async propose(input) { return this.createEvaluationPlan(input); }
  async execute(input) { return this.runBoundedChecks(input); }
  async evaluate(input) { return this.scoreClaims(input); }
  async produceEvidence(input) { return this.sealArtifacts(input); }
  async verifyEvidence(input) { return this.verifyProvenance(input); }
}

register(new EnterpriseEvaluatorAdapter({
  authority: 'VALIDATION_ONLY',
  externalActions: 0,
  settlementAuthority: 'NONE_GRANTED'
}));
